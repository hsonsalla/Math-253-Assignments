Task 2

Matrix Compositions

A = chol(Sigma)
A
##      [,1]       [,2]
## [1,]    1 -0.7000000
## [2,]    0  0.7141428
t(A)
##      [,1]      [,2]
## [1,]  1.0 0.0000000
## [2,] -0.7 0.7141428
# Matrix product of t(A) and A is equivalent to sigma.
all(abs((t(A) %*% A - Sigma)) < 0.01)
