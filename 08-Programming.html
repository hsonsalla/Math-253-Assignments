Task 1

x = runif(100, 20, 70)
y = 5 + 3*x + 2*rnorm(100)
My_data = data.frame(x,y)
plot(My_data)

LL_line = function(params){
  m = params[1]
  b = params[2]
  sigma = params[3]
  sum(log(dnorm(y - (m * x + b), sd = sigma)))
}

testA = LL_line(c(3,5,2))
testB = LL_line(c(4,1,10))

starting_params <- c(4, 1, 10)
best <- optim(starting_params, LL_line, control=list(fnscale = -1))

Task 2

load(url("http://tiny.cc/dcf/Taxi_trips.rda"))

taxi_likelihood = function(arguments){
  base_fare = arguments[1]
  per_mile = arguments[2]
  params = arguments[3]
  waiting_fare <- Taxi_trips$fare_amount - (base_fare + per_mile*Taxi_trips$trip_distance)
  -sum(dexp(abs(waiting_fare)))
}

start_params = (c(2.00, 4.00, 1/3))
best2 <- nlm(taxi_likelihood, start_params)

plot(Taxi_trips$fare_amount, Taxi_trips$trip_distance)
abline(best2$estimate[1], best2$estimate[2])
